pipeline {

    agent any

    environment {

        // Java
        JAVA_HOME = "C:\\Program Files\\Java\\jdk-17"

        // Maven
        MAVEN_HOME = "C:\\ProgramData\\Jenkins\\.jenkins\\tools\\hudson.tasks.Maven_MavenInstallation\\Maven"

        // PATH
        PATH = "${JAVA_HOME}\\bin;${MAVEN_HOME}\\bin;${env.PATH}"

        // Tomcat
        TOMCAT_HOME = "D:\\Server\\apache-tomcat-11.0.22"

        TOMCAT_WEBAPPS_PATH = "${TOMCAT_HOME}\\webapps"

        TOMCAT_SERVICE_NAME = "Tomcat11"

        // Docker Folder
        DOCKER_PATH = "C:\\Users\\LAPTOP WORLD\\Desktop\\Docker\\docker-p04"
    }

    stages {

        stage('Cleanup Workspace') {

            steps {

                deleteDir()
            }
        }

        stage('Clone Repository') {

            steps {

                git branch: 'main',
                    credentialsId: 'github-credentials',
                    url: 'https://github.com/Abhish0717/Advance_Project-04.git'
            }
        }

        stage('Build WAR File') {

            steps {

                echo "Building WAR File..."

                bat """
                "%MAVEN_HOME%\\bin\\mvn.cmd" clean package ^
                -DskipTests ^
                -f ORSProject04\\pom.xml
                """
            }
        }

        stage('Remove Old Deployment') {

            steps {

                echo "Removing old WAR and extracted folder..."

                bat """
                if exist "${TOMCAT_WEBAPPS_PATH}\\ORSProject04" (
                    rmdir /S /Q "${TOMCAT_WEBAPPS_PATH}\\ORSProject04"
                )

                if exist "${TOMCAT_WEBAPPS_PATH}\\ORSProject04.war" (
                    del /Q "${TOMCAT_WEBAPPS_PATH}\\ORSProject04.war"
                )
                """
            }
        }

        stage('Copy WAR File') {

            steps {

                script {

                    def warFile = "ORSProject04\\target\\ORSProject04.war"

                    echo "Copying WAR to Tomcat..."

                    bat """
                    copy /Y "${warFile}" "${TOMCAT_WEBAPPS_PATH}\\ORSProject04.war"
                    """

                    echo "Copying WAR to Docker Folder..."

                    bat """
                    copy /Y "${warFile}" "${DOCKER_PATH}\\ORSProject04.war"
                    """
                }
            }
        }

        stage('Ensure Tomcat Service Exists') {

            steps {

                script {

                    def serviceExists = bat(
                        script: "sc query ${TOMCAT_SERVICE_NAME}",
                        returnStatus: true
                    )

                    if (serviceExists != 0) {

                        echo "Installing Tomcat Service..."

                        bat """
                        cd /d "${TOMCAT_HOME}\\bin"
                        service.bat install
                        """

                        sleep time: 10, unit: 'SECONDS'

                    } else {

                        echo "Tomcat Service Already Exists"
                    }
                }
            }
        }

        stage('Restart Tomcat') {

            steps {

                script {

                    def status = bat(
                        script: 'sc query Tomcat11 | find "RUNNING"',
                        returnStatus: true
                    )

                    if (status == 0) {

                        echo "Stopping Tomcat..."

                        bat 'sc stop Tomcat11'

                        sleep time: 10, unit: 'SECONDS'
                    }

                    echo "Starting Tomcat..."

                    bat 'sc start Tomcat11'

                    sleep time: 40, unit: 'SECONDS'
                }
            }
        }

        stage('Verify Deployment') {

            steps {

                bat '''
                curl --fail http://localhost:8080/ORSProject04
                '''
            }
        }
    }

    post {

        always {

            echo 'Cleaning Workspace...'

            deleteDir()
        }

        success {

            echo 'Deployment Successful'
        }

        failure {

            echo 'Deployment Failed'
        }
    }
}
